            document.querySelectorAll('.address-card').forEach(c => {
                c.classList.remove('selected', 'bg-green-50', 'border-green-500');
                c.classList.add('bg-gray-50');
            });
            card.classList.add('selected', 'bg-green-50');
            card.classList.remove('bg-gray-50');
        }

        // ===== Checkout =====
        function openCheckout() {
            if (cart.length === 0) {
                showToast('السلة فارغة');
                return;
            }
            
            document.getElementById('cartModal').classList.add('hidden');
            document.getElementById('checkoutModal').classList.remove('hidden');
            
            // تحديث ملخص الطلب
            const container = document.getElementById('checkout-items');
            container.innerHTML = cart.map(item => `
                <div class="flex justify-between">
                    <span>${item.name} ×${item.quantity}</span>
                    <span class="font-bold">${(item.price * item.quantity).toFixed(3)} ر.ع</span>
                </div>
            `).join('');
            
            // تحديث الأيقونة والعنوان
            const iconBg = document.getElementById('checkout-icon-bg');
            const icon = document.getElementById('checkout-icon');
            const typeLabel = document.getElementById('checkout-type');
            const addressLabel = document.getElementById('checkout-address');
            
            if (orderType === 'delivery') {
                iconBg.className = 'w-12 h-12 bg-delivery-500 rounded-full flex items-center justify-center text-white';
                icon.className = 'fas fa-motorcycle text-xl';
                typeLabel.textContent = 'توصيل للمنزل';
                addressLabel.textContent = 'منطقة السرحات، الخابورة';
            } else {
                iconBg.className = 'w-12 h-12 bg-brand-400 rounded-full flex items-center justify-center text-dark-900';
                icon.className = 'fas fa-store text-xl';
                typeLabel.textContent = 'استلام من الفرع';
                addressLabel.textContent = 'الخابورة - السرحات - مقابل مركز الشرطة';
            }
            
            updateCheckoutTotals();
        }

        function closeCheckout() {
            document.getElementById('checkoutModal').classList.add('hidden');
        }

        function updateCheckoutTotals() {
            const subtotal = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
            const tax = subtotal * 0.05;
            const total = subtotal + tax + (orderType === 'delivery' ? deliveryFee : 0);
            
            document.getElementById('checkout-subtotal').textContent = subtotal.toFixed(3) + ' ر.ع';
            document.getElementById('checkout-tax').textContent = tax.toFixed(3) + ' ر.ع';
            document.getElementById('checkout-total').textContent = total.toFixed(3) + ' ر.ع';
        }

        function selectDeliveryTime(type) {
            document.querySelectorAll('.delivery-time-btn').forEach(btn => {
                btn.classList.remove('border-brand-400', 'bg-brand-50', 'text-dark-900');
                btn.classList.add('border-gray-200', 'text-gray-600');
            });
            
            event.currentTarget.classList.remove('border-gray-200', 'text-gray-600');
            event.currentTarget.classList.add('border-brand-400', 'bg-brand-50', 'text-dark-900');
            
            if (type === 'scheduled') {
                document.getElementById('scheduled-time-selector').classList.remove('hidden');
            } else {
                document.getElementById('scheduled-time-selector').classList.add('hidden');
            }
        }

        function selectPaymentMethod(method) {
            if (method === 'bank') {
                document.getElementById('checkout-bank-qr').classList.remove('hidden');
            } else {
                document.getElementById('checkout-bank-qr').classList.add('hidden');
            }
        }

        function placeOrder() {
            document.getElementById('checkoutModal').classList.add('hidden');
            document.getElementById('trackingModal').classList.remove('hidden');
            
            // محاكاة تحديث حالة الطلب
            setTimeout(() => {
                showToast('طلبك جاهز للاستلام!');
            }, 5000);
        }

        function closeTracking() {
            document.getElementById('trackingModal').classList.add('hidden');
            cart = [];
            updateCartUI();
        }
